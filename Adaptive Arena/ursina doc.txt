Showing a Picture/Image

my_image = Entity(
    model='quad',           	  # 'quad' = flat rectangle (for 2D)
    texture='path/to/image.png',  # Your image file
    scale=(4, 3),          	  # Width, Height (in Ursina units)
    position=(0, 0, 0),    	  # X, Y, Z coordinates
    enabled=True           	  # True = visible, False = hidden
)

Moving a sprite
P = Entity()

# Go To
P.x = 5 / P.y = 5 / P.position(x,y,z)

# Glide
P.animate_x(5, duration=1) / P.animate_y(5,duration=1) / P.animate_position((x,y,z), duration=1) # seconds
